function randomImageFromSerie(serieName){
	let l = tools_getRandomInt(0, RawDatas._IMAGES[serieName].length);
	return RawDatas._IMAGES[serieName][l] ;
}


function randomSerieName( exclude , _IMAGES ){
	let _names = [] ;
	for(var cat in _IMAGES){
		if(  cat != exclude ){
			_names.push( cat )
		}
	}
	
	let l = tools_getRandomInt(0, _names.length);
	return _names[l] ;
	
}


function shuffle(a) {
    for (let i = a.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [a[i], a[j]] = [a[j], a[i]];
    }
    return a;
}
